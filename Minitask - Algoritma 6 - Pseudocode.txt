FUNCTION TotalBelanjaSetelahDiskon(totalPembelian)
   DECLARE grandTotal, diskon: float

  BEGIN
     IF totalPembelian > 1000000 THEN
	 diskon <- 0,1
     ELSE IF totalPembelian >= 500000 && totalPembelian <= 1000000 THEN
	 diskon <- 0,05
     ELSE THEN
         diskon <- 0,0
     END IF
     grandTotal = totalPembelian - (totalPembelian * diskon)		
     RETURN grandTotal

END FUNCTION
